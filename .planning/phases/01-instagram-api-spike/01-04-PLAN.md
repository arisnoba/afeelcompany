Create src/app/admin/instagram-test/page.tsx.

Architecture: Single file with two components:
1. `SyncButton` — 'use client' component at top of file, handles the Sync Now click
2. `InstagramTestPage` — default export, async Server Component, reads DB and renders grid

**IMPORTANT — Next.js 16 Server Component rules:**
- Do NOT add `'use client'` to the default export (it is a Server Component)
- The `SyncButton` child component has its own `'use client'` directive
- `getCachedFeed()` is called directly in the server component (no useEffect, no useState)

```tsx
'use client'

import { useState } from 'react'

function SyncButton() {
  const [status, setStatus] = useState<'idle' | 'loading' | 'done' | 'error'>('idle')
  const [message, setMessage] = useState('')

  async function handleSync() {
    setStatus('loading')
    setMessage('')
    try {
      const res = await fetch('/api/instagram/sync', { method: 'POST' })
      const json = await res.json()
      if (json.success) {
        setStatus('done')
        setMessage(`Synced ${json.data.synced} posts. Refresh page to see updates.`)
      } else {
        setStatus('error')
        setMessage(json.error ?? 'Sync failed')
      }
    } catch {
      setStatus('error')
      setMessage('Network error during sync')
    }
  }

  return (
    <div style={{ display: 'flex', alignItems: 'center', gap: '12px', marginBottom: '24px' }}>
      <button
        onClick={handleSync}
        disabled={status === 'loading'}
        style={{
          padding: '8px 16px',
          background: status === 'loading' ? '#ccc' : '#000',
          color: '#fff',
          border: 'none',
          cursor: status === 'loading' ? 'not-allowed' : 'pointer',
          borderRadius: '4px',
        }}
      >
        {status === 'loading' ? 'Syncing...' : 'Sync Now'}
      </button>
      {message && (
        <span style={{ color: status === 'error' ? 'red' : 'green', fontSize: '14px' }}>
          {message}
        </span>
      )}
    </div>
  )
}
```

```tsx
// IMPORTANT: The 'use client' directive above applies ONLY to SyncButton.
// The default export below is a Server Component (no directive needed — server is default).

import Image from 'next/image'
import { getCachedFeed } from '@/lib/instagram'
import type { CachedPost } from '@/types/instagram'

export default async function InstagramTestPage() {
  const posts: CachedPost[] = await getCachedFeed()
  const lastSync = posts[0]?.fetched_at ?? null

  return (
    <div style={{ padding: '24px', maxWidth: '1200px', margin: '0 auto' }}>
      <h1 style={{ fontSize: '24px', fontWeight: 'bold', marginBottom: '8px' }}>
        Instagram Feed — Spike Test
      </h1>
      <p style={{ color: '#666', marginBottom: '16px', fontSize: '14px' }}>
        {lastSync
          ? `Last synced: ${new Date(lastSync).toLocaleString('ko-KR')}`
          : 'No data yet — click Sync Now to fetch from API'}
      </p>

      <SyncButton />

      {posts.length === 0 ? (
        <div style={{ padding: '48px', textAlign: 'center', color: '#999', border: '1px dashed #ccc', borderRadius: '8px' }}>
          <p>No cached posts found.</p>
          <p style={{ fontSize: '14px', marginTop: '8px' }}>
            Click &quot;Sync Now&quot; above to fetch from Instagram API.
          </p>
        </div>
      ) : (
        <div
          style={{
            display: 'grid',
            gridTemplateColumns: 'repeat(3, 1fr)',
            gap: '16px',
          }}
        >
          {posts.map((post) => (
            <div
              key={post.post_id}
              style={{
                border: '1px solid #e5e7eb',
                borderRadius: '8px',
                overflow: 'hidden',
                background: '#fff',
              }}
            >
              <div style={{ position: 'relative', aspectRatio: '1 / 1' }}>
                <Image
                  src={post.media_url}
                  alt={post.caption ?? 'Instagram post'}
                  fill
                  style={{ objectFit: 'cover' }}
                  sizes="(max-width: 768px) 100vw, 33vw"
                />
              </div>
              {post.caption && (
                <div style={{ padding: '12px', fontSize: '13px', color: '#374151' }}>
                  <p style={{
                    overflow: 'hidden',
                    display: '-webkit-box',
                    WebkitLineClamp: 3,
                    WebkitBoxOrient: 'vertical',
                  }}>
                    {post.caption}
                  </p>
                </div>
              )}
            </div>
          ))}
        </div>
      )}
    </div>
  )
}
```
